404: Page Not Found

It looks like you found a page that doesn't exist! Here's a gif, and the code behind it, as a consolation prize before heading home. 


document.addEventListener('DOMContentLoaded', function () {
	q = "finger guns"; // search query

	request = new XMLHttpRequest;
	request.open('GET', 'https://api.giphy.com/v1/gifs/random?api_key=dc6zaTOxFJmzC&rating=pg&tag=arrested-development+404-not-found'+q, true);

	request.onload = function() {
		if (request.status >= 200 && request.status < 400){
			data = JSON.parse(request.responseText).data.image_url;
			console.log(data);
			document.getElementById("giphyme").innerHTML = '<img src = "'+data+'"  title="404 Gif - Powered by GIPHY">';
		} else {
			console.log('reached giphy, but API returned an error');
		 }
	};

	request.onerror = function() {
		console.log('connection error');
	};

	request.send();
});
